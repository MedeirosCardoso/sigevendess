title Diagrama de Classes

abstract class Mercadoria{
  -codigo: Integer
  -descricao: String
  -categoria: CategoryTypes
  -simbolo: MeasureUnits
  -preco: float
  -estoqueAtual: float
}

class Produto{
  -receita: Receita
}

class Componente{
  -estoqueMin: float
}

class Receita{
  -codigo: Integer
  -produto: Produto
  -rendimento: float
  -tempoPreparo: float
  -componentes: List<ItemReceita>
  +addComponente(Componente, float): void
  +removeComponente(Componente): void
}

class ItemReceita{
  -receita: Receita
  -componente: Componente
  -qtdUtilizada: float
}

class Aquisicao{
    -codigo: Integer
    -componentes: List<itemAquisicao>
    -dataAquisicao: Date
    +addComponente(Componente, float, float): void
    +removeComponente(Componente): void
    +atualizarEstoqueComponente(): void
}

class ItemAquisicao{
    -aquisicao: Aquisicao
    -componente: Componente
    -qtdAdquirida: float
    -custo: float
}

class Telefone{
    -ddd: byte
    -digito: int
}

class Endereco{
    -codigo: Integer
    -cep: int
    -logradouro: String
    -numero: Short
    -bairro: String
    -complemento: String
    -cidade: String
    -estado: String
}

class Cliente{
    -numero: Telefone
    -nome: String
    -email: String
    -nomEstabelecimento: String
    -observacao: String
    -endereco: Endereco
}

class Pedido{
  -codigo: Integer
  -cliente: Cliente
  -situacao: Status
  -dataSolicitado: Date
  -dataEntrega: Date
  -dataProduzido: Date
  -itens: List<ItemPedido>
  -vlrTotalDesconto: float
  -vlrTotal: float
  -venda: Venda
  +addItem(Produto, float): void
  +removeItem(Produto): void
  +calcularVlrTotalPedido(): void
}

class ItemPedido{
    -pedido: Pedido
    -produto: Produto
    -qtdSolicitado: int
    -vlrUnit: float
    -vlrDescUnit: float
    +calcularVlrTotalItem(): float
}

class Venda{
    -codigo: Integer
    -opcPagamento: PaymentOptions
    -data: Date
    -vlrTotalDesc: float
    -vlrTotalVenda: float
    -pedidos: List<Pedido>
    +addPedido(Pedido): void
    +removePedido(Pedido): void
}

enum CategoryTypes <<enumeration>>{
    INGREDIENTE
    EMBALAGEM
    DOCE
    SALGADO
}

enum MeasureUnits <<enumeration>>{
    GRAMAS
    MILILITROS
    UNIDADE
    CENTO
}

enum Status <<enumeration>>{
    PRODUCAO
    PRONTO
    VENDIDO
    CANCELADO
}

enum PaymentOptions <<enumeration>>{
    DINHEIRO
    CREDITO
    DEBITO
    PRAZO
}

Produto o-- Receita
Componente --|> Mercadoria
Produto --|> Mercadoria
Mercadoria ..> CategoryTypes
Mercadoria ..> MeasureUnits
Receita ..> Componente
Receita *-- ItemReceita
Componente <--o ItemReceita
Componente <--o ItemAquisicao
Aquisicao *-- ItemAquisicao
Aquisicao ..> Componente
Cliente *--> Telefone
Cliente o--> Endereco
Pedido --> Cliente
Pedido *--ItemPedido
Pedido ..> Status
Pedido ..> Produto
ItemPedido o--> Produto
Venda ..> PaymentOptions
Venda *-- Pedido
Venda ..> Pedido

hide <<enumeration>> circle
@enduml
