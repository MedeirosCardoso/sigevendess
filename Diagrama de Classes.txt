title Diagrama de Classes

class Produto{
  -codigo: Integer
  -descricao: String
  -categoria: CategoryTypes
  -simbolo: MeasureUnits
  -preco: float
  -receita: Receita
}

class Componente{
  -estoqueMin: float
  -estoqueAtual: float
}

class Receita{
  -codigo: Integer
  -produto: Produto
  -rendimento: float
  -tempoPreparo: float
  -componentes: List<ItemReceita>
  +addComponente(Componente, float): void
  +removeComponente(Componente): void
}

class ItemReceita{
  -receita: Receita
  -componente: Componente
  -qtdUtilizada: float
}

class Aquisicao{
    -codigo: Integer
    -componentes: List<itemAquisicao>
    -dataAquisicao: Date
}

class ItemAquisicao{
    -aquisicao: Aquisicao
    -componente: Componente
    -qtdAdquirida: float
    -custo: float
}

class Telefone{
    -ddd: byte
    -digito: int
}

class Endereco{
    -codigo: Integer
    -cep: int
    -logradouro: String
    -numero: Short
    -bairro: String
    -complemento: String
    -cidade: String
    -estado: String
}

class Cliente{
    -numero: Telefone
    -nome: String
    -email: String
    -nomEstabelecimento: String
    -observacao: String
    -endereco: Endereco
}

class Pedido{
  -codigo: Integer
  -cliente: Cliente
  -situacao: Status
  -dataSolicitado: Date
  -dataEntrega: Date
  -itens: List<ItemPedido>
  -vendaPedido: Venda
}

class ItemPedido{
    -pedido: Pedido
    -produto: Produto
    -qtdSolicitado: int
    -vlrDescItem: float
    -vlrVendaitem: float
    -produzidoEm: Date
}

class Venda{
    -codigo: Integer
    -opcPagamento: PaymentOptions
    -data: Date
    -vlrTotalVenda: float
    -vlrTotalDesc: float
    -pedidoVenda: Pedido
}

enum CategoryTypes <<enumeration>>{
    INGREDIENTE
    EMBALAGEM
    DOCE
    SALGADO
}

enum MeasureUnits <<enumeration>>{
    GRAMAS
    MILILITROS
    UNIDADE
    CENTO
}

enum Status <<enumeration>>{
    PRODUCAO
    PRONTO
    VENDIDO
    CANCELADO
}

enum PaymentOptions <<enumeration>>{
    DINHEIRO
    CREDITO
    DEBITO
    PRAZO
}

Produto o-- Receita
Produto <|-- Componente
Produto ..> CategoryTypes
Produto ..> MeasureUnits
Receita ..> Componente
Receita *-- ItemReceita
Componente <--o ItemReceita
Componente <--o ItemAquisicao
Aquisicao *-- ItemAquisicao
Cliente *--> Telefone
Cliente o--> Endereco
Pedido --> Cliente
Pedido *--ItemPedido
Pedido ..> Status
ItemPedido o--> Produto
Venda ..> PaymentOptions
Venda *-- Pedido

hide <<enumeration>> circle
@enduml
