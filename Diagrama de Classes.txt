title Sigevendees Diagrama de Classes

class Produto {
  -Integer codigo
  -String descricao
  -CategoryTypes categoria
  -MeasureUnits simbolo
  -float preco
  -Receita receita
}

class Componente{
  -float estoqueMin
  -float estoqueAtual
}

class Receita{
  -Integer codigo
  -Produto produto
  -float rendimento
  -float tempoPreparo
  -List<ItemReceita> componentes
  +addComponente(Componente, float): void
  +removeComponente(Componente): void
}

class ItemReceita{
  -Receita receita
  -Componente componente
  -float qtdUtilizada
}

class Aquisicao{
    -Integer codigo
    -List<itemAquisicao> componentes
    -Date dataAquisicao
}

class ItemAquisicao{
    -Aquisicao aquisicao
    -Componente componente
    -float qtdAdquirida
    -float custo
}

class Telefone{
    -byte ddd
    -int digito
}

class Endereco{
    -Integer codigo
    -int cep
    -String logradouro
    -Short numero
    -String bairro
    -String complemento
    -String cidade
    -String estado
}

class Cliente{
    -Telefone numero
    -String nome
    -String email
    -String nomEstabelecimento
    -String observacao
    -Endereco endereco
}

class Pedido{
  -Integer codigo
  -Cliente cliente
  -Status situacao
  -Date dataSolicitado
  -Date dataEntrega
  -List<ItemPedido> itens
  -Venda venda
}

class ItemPedido{
    -Pedido pedido
    -Produto produto
    -int qtdSolicitado
    -float vlrTotDescItem
    -float vlrTotVendaitem
    -int qtdVendido
    -int qtdDescartado
    -Date produzidoEm
}

class Venda{
    -Integer codigo
    -PaymentOptions opcPagamento
    -Date data
    -float vlrTotalVenda
    -float vlrTotalDescVenda
    -Pedido pedido
}

enum CategoryTypes <<enumeration>>{
    INGREDIENTE
    EMBALAGEM
    DOCE
    SALGADO
}

enum MeasureUnits <<enumeration>>{
    GRAMAS
    MILILITROS
    UNIDADE
    CENTO
}

enum Status <<enumeration>>{
    PRODUCAO
    PRONTO
    VENDIDO
    CANCELADO
}

enum PaymentOptions <<enumeration>>{
    DINHEIRO
    CREDITO
    DEBITO
    PRAZO
}

Produto o-- Receita
Produto <|-- Componente
Produto ..> CategoryTypes
Produto ..> MeasureUnits
Receita ..> Componente
Receita *-- ItemReceita
Componente <--o ItemReceita
Componente <--o ItemAquisicao
Aquisicao *-- ItemAquisicao
Cliente *--> Telefone
Cliente o--> Endereco
Pedido --> Cliente
Pedido *--ItemPedido
Pedido ..> Status
ItemPedido o--> Produto
Venda ..> PaymentOptions
Venda *-- Pedido
